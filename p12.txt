#include<stdio.h>

int x[20],n;

int safe(int r,int c){
    for(int i=1;i<c;i++)
        if(x[i]==r || (x[i]-i)==(r-c) || (x[i]+i)==(r+c))
            return 0;
    return 1;
}

int queen(int c){
    if(c>n) return 1;
    for(int r=1;r<=n;r++){
        if(safe(r,c)){
            x[c]=r;
            if(queen(c+1)) return 1;
        }
    }
    return 0;
}

int main(){
    int i,j;
    printf("Enter number of queens: ");
    scanf("%d",&n);

    if(!queen(1)){
        printf("Solution does not exist");
        return 0;
    }

    for(i=1;i<=n;i++){
        for(j=1;j<=n;j++)
            printf(x[j]==i?"Q ":"# ");
        printf("\n");
    }
    return 0;
}


Output :

Q # # #
# # # Q
# Q # #
