#include<stdio.h>
#define INF 999

int main()
{
    int a[10][10],d[10],v[10]={0},n,s,i,j,min,u;

    printf("Enter n value: ");
    scanf("%d",&n);

    printf("Enter the graph data:\n");
    for(i=1;i<=n;i++)
        for(j=1;j<=n;j++)
            scanf("%d",&a[i][j]);

    printf("Enter the source node: ");
    scanf("%d",&s);

    for(i=1;i<=n;i++)
    {
        d[i]=a[s][i];
        v[i]=0;
    }
    v[s]=1;
    d[s]=0;

    for(i=1;i<n;i++)
    {
        min=INF;
        for(j=1;j<=n;j++)
            if(!v[j] && d[j]<min)
            {
                min=d[j];
                u=j;
            }

        v[u]=1;

        for(j=1;j<=n;j++)
            if(!v[j] && d[u]+a[u][j]<d[j])
                d[j]=d[u]+a[u][j];
    }

    for(i=1;i<=n;i++)
        printf("Shortest distance from %d to %d is %d\n",s,i,d[i]);

    return 0;
}
